Each

Proses perulangan di Eng2Js bisa dilakukan didalam syntax html, dan di berikan dengan prefix "each"

Contoh Each : 

eng.controller('ctrl', function(){
    this.data = [
        { nama: "budi", alamat:"depok" },
        { nama: "tono", alamat:"bogor" },
        { nama: "ridwan", alamat:"jakarta" },
        { nama: "garong", alamat:"bekasi" }
    ];
});
