Factory

Factory memungkinkan penggabungan object yang buat dari beberapa methode dan di package menjadi satu fungsi, ini memudahkan ketika kita memiliki banyak sekali konfigurasi dan modular dalam sebuah applikasi.

Contoh Factory : 
    
eng
.config('$cfg', { name:'ukung', count:0 })
.collection('$colect', { name:'string' })
.service('$srv', function(){
    return {
        combineName: function(name1, name2){
            return name1 + name2;
        } 
    }
})
.module('$mdl', function($root){
    this.balikNama = function(name){
        return name.split('').reverse().join('');
    };
})
.factory('$fact', function($root, $cfg, $srv, $mdl, $colect){
    return {
        balik: function(){
            var text = $mdl.balikNama( $srv.combineName($cfg.name, $root.name) );
            text += $cfg.count++;
            $colect.insert([ text ]);
        },
        store: $colect.store
    };
})
.controller('ctrl', function($fact){
    this.name = 'zulfah';
    this.testList = [];
    this.setName = function(){
        this.name = $fact.balik();
        this.name = 'zulfah';
        this.testList = $fact.store;
    };
});
